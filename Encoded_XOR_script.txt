-- 📜 Pastebin から暗号化スクリプトを取得

print("🔹 Encrypted Script:", encryptedScript) -- ✅ 暗号化スクリプトデバッグ

-- 🔓 XOR 復号関数
local function xor_decrypt(data, key)
    local result, keyLength = {}, #key
    for i = 1, #data do
        result[i] = string.char(bit32.bxor(data:byte(i), key:byte((i - 1) % keyLength + 1)))
    end
    return table.concat(result)
end

-- 🚀 復号＆実行
local finalScript = xor_decrypt(encryptedScript, key)
print("🔓 Decrypted Script:", finalScript) -- ✅ 復号スクリプトデバッグ

-- エラー対策
assert(finalScript and #finalScript > 0, "❌ Decryption failed: Script is empty!")
